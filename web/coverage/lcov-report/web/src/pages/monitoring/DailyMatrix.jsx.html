export const DailyMatrixRow = ({ user, boxClass, currentUser }) => {

    currentUser &&
    (user.userId === currentUser.id || user.nama === currentUser.nama);
 
  return (
    <tr

            day

        >

    </tr>
  );
};
 
const DailyMatrix = ({ data = [], monthIndex, year }) => {
